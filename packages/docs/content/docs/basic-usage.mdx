import { MentisDemo } from "@/components";

```tsx
import { MentionInput } from "mentis";
import "mentis/dist/index.css";

function App() {
  const [value, setValue] = useState<string>("");
  return (
    <MentionInput
      defaultValue={value}
      options={[
        { label: "Alice", value: "alice" },
        { label: "Bob", value: "bob" },
        { label: "Charlie", value: "charlie" },
      ]}
      onChange={setValue}
    />
  );
}
```
